#include IRremote.h
IRsend irsend;
const int pin1 = 10;
const int pin2 = 9;
const int pin3 = 8;
const int pin4 = 5;
const int pin5 = 7;
const int pin6 = 6;
int val1;
int val2;
int val3;
int val4;
int val5;
int val6;
void setup() {
Serial.begin(9600);
Serial.print("Starting");
pinMode(pin1, INPUT);
pinMode(pin2, INPUT);
pinMode(pin3, INPUT);
pinMode(pin4, INPUT);
pinMode(pin5, INPUT);
pinMode(pin6, INPUT);
}
void loop() {
val1 = 0;
val2 = 0;
val3 = 0;
val4 = 0;
val5 = 0;
val6 = 0;
val1 = digitalRead(pin1);
val2 = digitalRead(pin2);
val3 = digitalRead(pin3);
val4 = digitalRead(pin4);
val5 = digitalRead(pin5);
val6 = digitalRead(pin6);
Serial.print("val1 : ");
Serial.print(val1);
Serial.print("/ ");
Serial.print("val2 : ");
Serial.print(val2);
Serial.print("/ ");
Serial.print("val3 : ");
Serial.print(val3);
Serial.print("/ ");
Serial.print("val4 : ");
Serial.print(val4);
Serial.print("/ ");
Serial.print("val5 : ");
Serial.print(val5);
Serial.print("/ ");
Serial.print("val6 : ");
Serial.print(val6);
Serial.println();
delay(100);
if (val1 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA09F6, 32);
Serial.println("on/off");
delay(40);
}
}
else if (val6 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA49B6, 32);
Serial.println("freeze");
delay(40);
}
}
else if (val2 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA11EE, 32);
Serial.println("zoom up");
delay(40);
}
}
else if (val3 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA916E, 32);
Serial.println("zoom down");
delay(40);
}
}
else if (val4 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA19E6, 32);
Serial.println("vol up");
delay(40);
}
}
else if (val5 == HIGH){
for (int i = 0; i < 3; i++) {
irsend.sendNEC(0xC1AA9966, 32);
Serial.println("vol down");
delay(40);
}
}
}
